大多数针对字符串或数组的indexOf调用的检查都将其与-1进行比较，因为0是有效的索引。 如果您只是检查字符串的存在，请考虑改用includes。

不合规示例

var color = "blue";
var name = "ishmael";
var number = 123;

var arr = [color, name];

if (arr.indexOf("blue") > 0) { // 不合规
  // ...
}
if (name.indexOf("ish") > 0) { // 不合规
  // ...
}

合规示例

var color = "blue";
var name = "ishmael";
var number = 123;

var arr = [color, name];

if (arr.indexOf("blue") >= 0) {
  // ...
}
if (name.includes("ish")) {
  // ...
}
