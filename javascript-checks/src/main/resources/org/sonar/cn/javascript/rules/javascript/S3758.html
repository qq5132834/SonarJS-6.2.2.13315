NaN不等任务值，甚至不等于自己。所以当比较的一方是 NaN 时，返回的值总是 false。 总之undefined，null，NaN 和无法转换为数字的对象在用于数值比较时的计算结果为 NaN。

不合规示例

var x;  // x 的值是 undefined
if (someCondition()) {
  x = 42;
}

if (42 > x) {  // 不合规，x的值可能还是 undefined
  doSomething();
}

var obj = {prop: 42};
if (obj > 24) { // 不合规
  doSomething();
}

合规示例

var x;
if (someCondition()) {
  x = 42;
} else {
  x = foo();
}

if (42 > x) {
  doSomething();
}

var obj = {prop: 42};
if (obj.prop > 24) {
  doSomething();
}
