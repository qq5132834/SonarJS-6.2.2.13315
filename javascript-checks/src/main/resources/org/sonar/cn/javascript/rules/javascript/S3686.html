创建新对象实例的构造函数只能用 new 调用。 非构造函数不能。 混合使用这两种用法可能会在运行时导致意外结果。

不合规示例

function getNum() {
  return 5;
}

function Num(numeric, alphabetic) {
  this.numeric = numeric;
  this.alphabetic = alphabetic;
}

var myFirstNum = getNum();
var my2ndNum = new getNum();  // 不合规，空对象返回，不是5

var myNumObj1 = new Num();
var myNumObj2 = Num();  // 不合规，返回undefined，不是一个对象
