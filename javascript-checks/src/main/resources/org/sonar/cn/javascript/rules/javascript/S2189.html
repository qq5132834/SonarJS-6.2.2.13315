无限循环是一个在程序运行时永远不会结束的循环，必须通过终止程序才能退出循环。 无论是通过满足循环的结束条件还是通过break，每个循环都应该有一个结束条件。

不合规示例

for (;;) {  // 不合规，未设置循环结束条件
  // ...
}

var j = 0;
while (true) { // 不合规，死循环
  j++;
}

var k;
var b = true;
while (b) { // 不合规，死循环
  k++;
}

合规示例


while (true) { // 中断将可能允许离开循环
  if (someCondition) {
    break;
  }
}

var k;
var b = true;
while (b) {
  k++;
  b = k < 10;
}

outer:
while(true) {
  while(true) {
    break outer;
  }
}
