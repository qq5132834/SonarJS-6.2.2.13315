不应使用重复元素定义并集和交集类型

不合规示例

function padLeft(value: string, padding: string | number | string) { // 不合规； 'string' 类型在联合类型声明中使用了两次
  // ...
}

function extend(p : Person) : Person & Person & Loggable { // 不合规，Person是两次使用
 // ...
}

合规示例

function padLeft(value: string, padding: string | number | boolean) {
  // ...
}

function extend(p : Person) : Person & Loggable {
  // ...
}
