让 switch 或 if 链中的所有分支具有相同的实现是错误的。

不合规示例

if (b == 0) {  // 不合规
  doOneMoreThing();
}
else {
  doOneMoreThing();
}

let a = (b == 0) ? getValue() : getValue();   // 不合规

switch (i) {  // 不合规
  case 1:
    doSomething();
    break;
  case 2:
    doSomething();
    break;
  case 3:
    doSomething();
    break;
  default:
    doSomething();
}
