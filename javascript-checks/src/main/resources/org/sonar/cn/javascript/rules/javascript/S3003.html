对字符串使用比较运算符不太可能产生预期的结果。

不合规示例

var appleNumber = "123";
var orangeNumber = "45";
if (appleNumber < orangeNumber) {  // 不合规，条件为true
  alert("There are more oranges");
}

合规示例

var appleNumber = "123";
var orangeNumber = "45";
if (Number(appleNumber) < Number(orangeNumber)) {
  alert("There are more oranges");
}
