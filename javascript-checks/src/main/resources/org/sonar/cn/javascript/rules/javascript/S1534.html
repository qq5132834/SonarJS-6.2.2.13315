JavaScript 允许在类和对象文字中使用重复的属性名称，但只有重复名称的最后一个实例才能确定将用于它的实际值。 因此，更改其他出现的重复名称的值不会有任何效果，并且可能会导致误解和错误。

使用重复的constructor 定义class 将产生错误。

在 ECMAScript 2015 之前，使用重名会在 JavaScript 严格模式代码中产生错误。

不合规示例

var data = {
  "key": "value",
  "1": "value",
  "key": "value", // 不合规 - "key" 的副本
  'key': "value", // 不合规 - "key" 的副本
  key: "value", // 不合规 - "key" 的副本
  \u006bey: "value", // 不合规 - "key" 的副本
  "\u006bey": "value", // 不合规 - "key" 的副本
  "\x6bey": "value", // 不合规 - "key"  的副本
  1: "value" // 不合规 - "1" 的副本
}

合规示例

var data = {
  "key": "value",
  "1": "value",
  "key2": "value",
  'key3': "value",
  key4: "value",
  \u006bey5: "value",
  "\u006bey6": "value",
  "\x6bey7": "value",
  1b: "value"
}
