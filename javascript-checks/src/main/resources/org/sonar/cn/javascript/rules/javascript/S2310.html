不应在循环体中修改计数器。

不合规示例

var names = [ "Jack", "Jim", "", "John" ];
for (var i = 0; i < names.length; i++) {
  if (!names[i]) {
    i = names.length;                                 // 不合规
  } else {
    console.log(names[i]);
  }
}

合规示例

var names = [ "Jack", "Jim", "", "John" ];
for (var name of names) {
  if (!name) {
    break;                                 // 合规
  } else {
    console.log(name);
  }
}
