最多迭代一次的循环相当于使用if语句有条件地执行一段代码。 没有开发人员期望找到循环语句的这种用法。 如果作者的初衷真的是有条件地执行一段代码，那么应该使用if语句来代替。

在最坏的情况下，这不是作者的初衷，因此循环体应该固定为使用嵌套的 return、break 或 throw 以更恰当的方式陈述。

不合规示例

for (let i = 0; i < 10; i++) { // 不合规, 循环只执行一次
  console.log("i is " + i);
  break;
}
...
for (let i = 0; i < 10; i++) { // 不合规, 循环只执行一次
  if (i == x) {
    break;
  } else {
    console.log("i is " + i);
    return;
  }
}

合规示例

for (let i = 0; i < 10; i++) {
  console.log("i is " + i);
}
...
for (let i = 0; i < 10; i++) {
  if (i == x) {
    break;
  } else {
    console.log("i is " + i);
  }
}
