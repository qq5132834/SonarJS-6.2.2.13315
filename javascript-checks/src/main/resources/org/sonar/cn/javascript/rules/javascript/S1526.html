使用 var 声明的变量具有特殊属性，无论它们在函数中的何处声明，它们都会“浮动”到函数的顶部，甚至在声明之前就可以使用。 这使得范围界定变得混乱，尤其是对于新编码员而言。

为了将混淆降到最低，var 声明应该在它们第一次使用之前进行。

不合规示例

var x = 1;

function fun(){
  alert(x); // 不合规，因为稍后在同一范围内声明了 x
  if(something) {
    var x = 42; // 函数范围内的声明（不是块范围！）隐藏全局变量
  }
}

fun(); // 警告是"undefined"而不是"1"

合规示例

var x = 1;

function fun() {
  print(x);
  if (something) {
    x = 42;
  }
}

fun(); // 打印 "1"
