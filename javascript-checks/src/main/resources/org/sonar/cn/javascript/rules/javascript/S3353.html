为了代码清晰起见，不更新的变量应标记为 const。

不合规示例

function seek(input) {
  let target = 32;  // 不合规
  for (let i of input) {
    if (i == target) {
      return true;
    }
  }
  return false;
}

function getUrl(query) {    
  let url; // 不合规
  url = "http://example.com";
  return url;
}

合规示例

function seek(input) {
  const target = 32;
  for (let i of input) {
    if (i == target) {
      return true;
    }
  }
  return false;
}

function getUrl(query) {  
  const url = "http://example.com";
  return url;
}
