在块内部声明但在块外部使用的变量（可以使用var声明）应该在块外部声明。

不合规示例

function doSomething(a, b) {
  if (a > b) {
    var x = a - b;  // 不合规
  }

  if (a > 4) {
    console.log(x);
  }

  for (var i = 0; i < m; i++) { // 不合规, 两个循环使用相同的变量
  }

  for (var i = 0; i < n; i++) {
  }

  return a + b;
}

合规示例

function doSomething(a, b) {
  var x;

  if (a > b) {
    x = a - b;
  }

  if (a > 4) {
    console.log(x);
  }

  for (let i = 0; i < m; i++) {
  }

  for (let i = 0; i < n; i++) {
  }


  return a + b;
}
