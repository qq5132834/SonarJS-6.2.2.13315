ECMAScript 2015 引入了同时从对象或数组中提取和分配多个数据点的功能。 这称为解构，它允许您压缩样板代码，以便您可以专注于逻辑。

当从同一对象或数组中提取多条数据并将其分配给变量时，此规则会引发问题。

不合规示例

function foo (obj1, obj2, array) {
  var a = obj1.a;  // 不合规
  var b = obj1.b;

  var name = obj2.name;  // 忽略； 只有一个提取和分配

  var zero = array[0];  // 不合规
  var one = array[1];
}

合规示例

function foo (obj1, obj2, array) {
  var {a, b} = obj1;

  var {name} = obj2;  // 此语法有效，因为 var 名称和属性名称相同

  var [zero, one] = array;
}
