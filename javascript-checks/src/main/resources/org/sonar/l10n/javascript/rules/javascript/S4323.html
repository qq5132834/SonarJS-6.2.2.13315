联合和交集类型很方便，但会使代码更难阅读和维护。 因此，如果在多个地方使用特定的并集或交集，建议使用类型别名。

不合规示例

function foo(x:string|null|number) { // 不合规
  // ...
}
function bar(x:string|null|number) {
  // ...
}
function zoo(): string|null|number {
  return null;
}

合规示例

type MyType = string | null | number;

function foo(x: MyType) {
  // ...
}
function bar(x: MyType) {
  // ...
}
function zoo(): MyType {
  return null;
}
