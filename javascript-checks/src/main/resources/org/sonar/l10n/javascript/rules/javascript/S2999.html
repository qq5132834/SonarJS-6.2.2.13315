new 关键字只能用于定义构造函数的对象。 将它与其他任何东西一起使用，您将得到一个 TypeError，因为没有要调用的 new 关键字的构造函数。

不合规示例

function MyClass() {
  this.foo = 'bar';
}

var someClass = 1;

var obj1 = new someClass;    // 不合规
var obj2 = new MyClass();    // 不合规

合规示例

/**
 * @constructor
 */
function MyClass() {
  this.foo = 'bar';
}

var someClass = function(){
  this.prop = 1;
}

var obj1 = new someClass;  // 合规
var obj2 = new MyClass();  // 合规
