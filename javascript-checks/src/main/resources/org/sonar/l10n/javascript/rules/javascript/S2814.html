同一名称多次用作变量或函数，但这样做可能会使维护人员感到困惑。

This rule also applies to function parameters.

不合规示例

var a = 'foo';
function a() {}   // 不合规
console.log(a);   // 输出 "foo"

function myFunc(arg) {
  var arg = "event"; // 不合规，参数值丢失
}

fun(); // 输出 "bar"

function fun() {
  console.log("foo");
}

fun(); // 输出 "bar"

function fun() {  // 不合规
  console.log("bar");
}

fun(); // 输出 "bar"

合规示例

var a = 'foo';
function otherName() {}
console.log(a);

function myFunc(arg) {
  var newName = "event";
}

fun(); // 输出 "foo"

function fun() {
  print("foo");
}

fun(); // 输出 "foo"

function printBar() {
  print("bar");
}

printBar(); // 输出 "bar"
