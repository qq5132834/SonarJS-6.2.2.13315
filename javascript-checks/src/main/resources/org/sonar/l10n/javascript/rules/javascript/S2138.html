不要给变量赋值为undefined，根据需要可以设置为null。

不合规示例

var myObject = {};

// ...
myObject.fname = undefined;  // 不合规
// ...

if (myObject.lname == undefined) {
  // 尚未创建属性
}
if (myObject.fname == undefined) {
  // 无法真正了解myObject.fname的真实状态
}

合规示例

var myObject = {};

// ...
myObject.fname = null;
// ...

if (myObject.lname == undefined) {
  // 尚未创建属性
}
if (myObject.fname == undefined) {
  // 无法真正了解myObject.fname的真实状态
}
