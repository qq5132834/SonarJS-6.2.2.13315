当对代码进行 null 或 undefined 相等判断和使用时，有可能导致出现TypeError问题。

不合规示例

if (str == null && str.length == 0) {
  console.log("String is empty");
}

if (str == undefined && str.length == 0) {
  console.log("String is empty");
}

if (str != null || str.length > 0) {
  console.log("String is not empty");
}

if (str != undefined || str.length > 0) {
  console.log("String is not empty");
}

合规示例

if (str != null && str.length == 0) {
  console.log("String is empty");
}

if (str != undefined && str.length == 0) {
  console.log("String is empty");
}

if (str == null || str.length > 0) {
  console.log("String is not empty");
}

if (str == undefined || str.length > 0) {
  console.log("String is not empty");
}
