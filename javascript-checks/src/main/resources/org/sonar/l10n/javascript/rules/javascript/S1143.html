break、continue、return 和 throw不应出现在finally块中。

不合规示例

function foo() {
    try {
        return 1; // 我们期望返回 1
    } catch(err) {
        return 2; // 在错误的情况下返回 2
    } finally {
        return 3; // 不合规: 3 在 1 或 2 之后返回，这真是匪夷所思
    }
}

合规示例

function foo() {
    try {
        return 1; // 我们期望返回 1
    } catch(err) {
        return 2; // 在错误的情况下返回 2
    }
}
