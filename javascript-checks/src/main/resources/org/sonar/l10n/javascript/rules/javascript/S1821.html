switch语句不要嵌套使用。虽然在语法上支持，但是不易于理解和维护。

不合规示例

function foo(n, m) {
  switch (n) {
    case 0:
      switch (m) {  // 不合规; switch嵌套
        // ...
      }
    case 1:
      // ...
    default:
      // ...
  }
}

合规示例

function foo(n, m) {
  switch (n) {
    case 0:
      bar(m);
    case 1:
      // ...
    default:
      // ...
  }
}

function bar(m) {
  switch(m) {
    // ...
  }
}
