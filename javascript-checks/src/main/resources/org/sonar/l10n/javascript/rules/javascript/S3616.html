逗号和逻辑 OR 运算符不应在 switch case 中使用

不合规示例

switch a {
  case 1,2:  // 不合规，仅仅处理case 2情况
    doTheThing(a);
  case 3 || 4: // 不合规，仅仅处理case 3情况
    doThatThing(a);
  case 5:
    doTheOtherThing(a);
  default:
    console.log("Neener, neener!");  //在a==1或4时会执行
}

合规示例

switch a {
  case 1:
  case 2:
    doTheThing(a);
  case 3:
  case 4:
    doThatThing(a);
  case 5:
    doTheOtherThing(a);
  default:
    console.log("Neener, neener!");
}
