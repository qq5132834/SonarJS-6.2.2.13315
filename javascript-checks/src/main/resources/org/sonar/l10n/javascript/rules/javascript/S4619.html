在数组上使用的 in 运算符在语法上是可行的，但是实际效果却不一定是预期的。

不合规示例

function func1() {
    let arr = ["a", "b", "c"];

    let expectedValue = "b";
    if (expectedValue in arr) { //不合规，总是false
        return expectedValue + " found in the array";
    } else {
        return expectedValue + " not found";
    }
}

function func2() {
    let arr = ["a", "b", "c"];

    let expectedValue = "1"; // 索引#1对应于值“b”
    if (expectedValue in arr) { // 不合规，将始终为真，因为数组由 3 个元素组成，并且无论其值如何，#1 始终存在
        return expectedValue + " found in the array";
    } else {
        return expectedValue + " not found";
    }
}

合规示例

function func() {
    let arr = ["a", "b", "c"];

    let expectedValue = "b";
    if (arr.includes(expectedValue)) {
        return expectedValue + " was found in the array";
    } else {
        return expectedValue + " not found";
    }
}
