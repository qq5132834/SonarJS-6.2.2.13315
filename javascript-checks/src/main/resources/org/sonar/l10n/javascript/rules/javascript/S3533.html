在 ECMAScript 2015 之前，模块管理必须是临时的或由 Node.js、Webpack 或 RequireJS 等第 3 方库提供。 幸运的是，ES2015 为模块管理、import 和 export 提供了语言标准机制。

不合规示例

// circle.js
exports.area = function (r) {
  return PI * r * r;
};

// foo.js
define(["./cart", "./horse"], function(cart, horse) {  // 不合规
  // ...
});

// bar.js
const circle = require('./circle.js');  // 不合规

合规示例

// circle.js
let area = function (r) {
  return PI * r * r;
}
export default area;

// foo.js
import cart from "./cart.js";
import horse from "./horse.js";

// bar.js
import circle from "./circle.js"
