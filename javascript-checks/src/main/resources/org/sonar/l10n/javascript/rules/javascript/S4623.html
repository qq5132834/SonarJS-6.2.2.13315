undefined不应作为可选参数的值传递

不合规示例

function foo(x: number, y: string = "default", z?: number) {
  // ...
}

foo(42, undefined); // 不合规
foo(42, undefined, undefined); // 不合规
foo(42, undefined, 5); // 没有其他方法可以强制第二个参数的默认值

合规示例

function foo(x: number, y: string = "default", z?: number) {
  // ...
}

foo(42);
