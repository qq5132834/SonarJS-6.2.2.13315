加密数据是安全敏感的。 它在过去导致了以下漏洞：

  • CVE-2017-7902
  • CVE-2006-1378
  • CVE-2003-1376

敏感示例

// === Client side ===
crypto.subtle.encrypt(algo, key, plainData); // 敏感
crypto.subtle.decrypt(algo, key, encData); // 敏感

// === Server side ===
const crypto = require("crypto");
const cipher = crypto.createCipher(algo, key); // 敏感
const cipheriv = crypto.createCipheriv(algo, key, iv); // 敏感
const decipher = crypto.createDecipher(algo, key); // 敏感
const decipheriv = crypto.createDecipheriv(algo, key, iv); // 敏感
const pubEnc = crypto.publicEncrypt(key, buf); // 敏感
const privDec = crypto.privateDecrypt({ key: key, passphrase: secret }, pubEnc); // 敏感
const privEnc = crypto.privateEncrypt({ key: key, passphrase: secret }, buf); // 敏感
const pubDec = crypto.publicDecrypt(key, privEnc); // 敏感
