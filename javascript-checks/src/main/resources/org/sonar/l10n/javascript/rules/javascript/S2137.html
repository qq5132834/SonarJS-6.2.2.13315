JavaScript具有特殊标识符，虽然没有保留，但仍不应用作标识符。 他们包括：

  • eval - 将字符串求值为JavaScript代码
  • arguments - 用于通过索引的财产访问函数参数。
  • undefined - 返回没有赋值的属性
  • NaN - 不是数字；当数学函数失败时返回
  • Infinity - 当数字超过浮点数的上限时

不应绑定或分配这些词，因为这样做会覆盖这些标识符的原始定义。 此外，分配或绑定其中一些名称将在JavaScript严格模式代码中生成错误。

不合规示例

eval = 17; // 不合规
arguments++; // 不合规
++eval; // 不合规
var obj = { set p(arguments) { } }; // 不合规
var eval; // 不合规
try { } catch (arguments) { } // 不合规
function x(eval) { } // 不合规
function arguments() { } // 不合规
var y = function eval() { }; // 不合规
var f = new Function("arguments", "return 17;"); // 不合规

function fun() {
  if (arguments.length == 0) { // 合规
    // do something
  }
}

合规示例

result = 17;
args++;
++result;
var obj = { set p(arg) { } };
var result;
try { } catch (args) { }
function x(arg) { }
function args() { }
var y = function fun() { };
var f = new Function("args", "return 17;");

function fun() {
  if (arguments.length == 0) {
    // do something
  }
}
