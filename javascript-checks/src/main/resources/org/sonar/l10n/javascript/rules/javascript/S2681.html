可以从单行块中省略大括号，例如使用if语句或for循环，但这样做可能会产生误导并导致错误。

不合规示例

if (condition)
  firstActionInBlock();
  secondAction();  // 不合规；无条件执行
thirdAction();

if (condition) firstActionInBlock(); secondAction();  // 不合规；secondAction无条件执行

if (condition) firstActionInBlock();  // 不合规
  secondAction();  // 无条件执行

if (condition); secondAction();  // 不合规； secondAction 无条件执行

let str = undefined;
for (let i = 0; i < array.length; i++)
  str = array[i];
  doTheThing(str);  // 不合规；仅在最后一个数组元素上执行

合规示例

if (condition) {
  firstActionInBlock();
  secondAction();
}
thirdAction();

let str = undefined;
for (let i = 0; i < array.length; i++) {
  str = array[i];
  doTheThing(str);
}
