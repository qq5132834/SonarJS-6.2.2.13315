arguments.caller 和 arguments.callee使许多优化变得不可能，因此在最新版本的JavaScript中，它们已被弃用。

不合规示例

function whoCalled() {
   if (arguments.caller == null)   //不合规
      console.log('I was called from the global scope.');
   else
      console.log(arguments.caller + ' called me!');  // 不合规

  console.log(whoCalled.caller);  // 不合规
  console.log(whoCalled.arguments);  // 不合规
}
