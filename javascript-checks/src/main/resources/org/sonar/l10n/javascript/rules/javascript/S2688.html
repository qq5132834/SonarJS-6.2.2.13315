NaN不等于任何东西，甚至不等于它本身。 根据NaN测试相等或不相等将产生可预测的结果，但可能不是您想要的结果。

不合规示例

var a = NaN;

if (a === NaN) {  // 不合规；总是false
  console.log("a is not a number");  // 这是死代码
}
if (a !== NaN) { // 不合规，总是为true
  console.log("a is not NaN"); // 这条语句不一定是true
}

合规示例

if (Number.isNaN(a)) {
  console.log("a is not a number");
}
if (!Number.isNaN(a)) {
  console.log("a is not NaN");
}
